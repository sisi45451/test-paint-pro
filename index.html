-- Profiles utilisateurs
CREATE TABLE profiles (
  id UUID REFERENCES auth.users ON DELETE CASCADE PRIMARY KEY,
  email TEXT,
  nom_entreprise TEXT,
  telephone TEXT,
  created_at TIMESTAMPTZ DEFAULT NOW()
);
ALTER TABLE profiles ENABLE ROW LEVEL SECURITY;
CREATE POLICY "Users can manage own profile"
  ON profiles FOR ALL USING (auth.uid() = id);

-- Clients
CREATE TABLE clients (
  id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
  user_id UUID REFERENCES auth.users ON DELETE CASCADE,
  nom TEXT NOT NULL,
  telephone TEXT,
  email TEXT,
  adresse TEXT,
  type TEXT DEFAULT 'particulier',
  created_at TIMESTAMPTZ DEFAULT NOW()
);
ALTER TABLE clients ENABLE ROW LEVEL SECURITY;
CREATE POLICY "Users manage own clients"
  ON clients FOR ALL USING (auth.uid() = user_id);

-- Devis
CREATE TABLE devis (
  id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
  user_id UUID REFERENCES auth.users ON DELETE CASCADE,
  client_id UUID REFERENCES clients ON DELETE SET NULL,
  numero TEXT,
  statut TEXT DEFAULT 'brouillon',
  tva_taux NUMERIC DEFAULT 10,
  acompte_pct NUMERIC DEFAULT 30,
  remarques TEXT,
  total_ht NUMERIC DEFAULT 0,
  total_tva NUMERIC DEFAULT 0,
  total_ttc NUMERIC DEFAULT 0,
  created_at TIMESTAMPTZ DEFAULT NOW()
);
ALTER TABLE devis ENABLE ROW LEVEL SECURITY;
CREATE POLICY "Users manage own devis"
  ON devis FOR ALL USING (auth.uid() = user_id);

-- Lignes de devis
CREATE TABLE devis_lignes (
  id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
  devis_id UUID REFERENCES devis ON DELETE CASCADE,
  designation TEXT,
  quantite NUMERIC DEFAULT 1,
  unite TEXT DEFAULT 'm²',
  prix_unitaire NUMERIC DEFAULT 0,
  total NUMERIC DEFAULT 0,
  ordre INT DEFAULT 0
);
ALTER TABLE devis_lignes ENABLE ROW LEVEL SECURITY;
CREATE POLICY "Users manage own lignes"
  ON devis_lignes FOR ALL USING (
    EXISTS (SELECT 1 FROM devis d WHERE d.id = devis_id AND d.user_id = auth.uid())
  );

-- Chantiers
CREATE TABLE chantiers (
  id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
  user_id UUID REFERENCES auth.users ON DELETE CASCADE,
  client_id UUID REFERENCES clients ON DELETE SET NULL,
  devis_id UUID REFERENCES devis ON DELETE SET NULL,
  nom TEXT NOT NULL,
  adresse TEXT,
  statut TEXT DEFAULT 'en_cours',
  avancement INT DEFAULT 0,
  date_debut DATE,
  date_fin_prevue DATE,
  montant NUMERIC DEFAULT 0,
  created_at TIMESTAMPTZ DEFAULT NOW()
);
ALTER TABLE chantiers ENABLE ROW LEVEL SECURITY;
CREATE POLICY "Users manage own chantiers"
  ON chantiers FOR ALL USING (auth.uid() = user_id);

-- Journal chantier
CREATE TABLE journal_chantier (
  id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
  chantier_id UUID REFERENCES chantiers ON DELETE CASCADE,
  note TEXT,
  avancement INT,
  created_at TIMESTAMPTZ DEFAULT NOW()
);
ALTER TABLE journal_chantier ENABLE ROW LEVEL SECURITY;
CREATE POLICY "Users manage own journal"
  ON journal_chantier FOR ALL USING (
    EXISTS (SELECT 1 FROM chantiers c WHERE c.id = chantier_id AND c.user_id = auth.uid())
  );

-- Trigger profile auto à l'inscription
CREATE OR REPLACE FUNCTION handle_new_user()
RETURNS TRIGGER AS $$
BEGIN
  INSERT INTO profiles (id, email)
  VALUES (NEW.id, NEW.email);
  RETURN NEW;
END;
$$ LANGUAGE plpgsql SECURITY DEFINER;

CREATE TRIGGER on_auth_user_created
  AFTER INSERT ON auth.users
  FOR EACH ROW EXECUTE FUNCTION handle_new_user();
